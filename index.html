#!/bin/bash RED='\033[0;31m' GREEN='\033[0;32m' YELLOW='\033[1;33m' BLUE='\033[0;34m' NC='\033[0m' clear echo -e "${BLUE}╔════════════════════════════════════════════════════════════╗${NC}" echo -e "${BLUE}║ 🚀 VPS MIGRATION TOOL v2 - SUPER OTOMATIS 🚀 ║${NC}" echo -e "${BLUE}║ Semua sudah otomatis! Tinggal reboot di akhir ║${NC}" echo -e "${BLUE}╚════════════════════════════════════════════════════════════╝${NC}" echo "" read -p "IP Address VPS Lama: " SOURCE_IP read -s -p "Password Root VPS Lama: " SOURCE_PASS echo "" read -p "Hostname baru untuk VPS ini (tekan Enter = migrated-vps): " NEW_HOST if [ -z "$NEW_HOST" ]; then NEW_HOST="migrated-vps"; fi echo "" echo -e "${GREEN}✅ Memeriksa prasyarat...${NC}" apt-get update -qq apt-get install -y sshpass rsync netplan.io > /dev/null 2>&1 echo -e "${YELLOW}🔍 Mencoba koneksi...${NC}" if ! sshpass -p "$SOURCE_PASS" ssh -o StrictHostKeyChecking=no root@$SOURCE_IP "echo OK" > /dev/null 2>&1; then echo -e "${RED}❌ Gagal konek!${NC}" unset SOURCE_PASS exit 1 fi echo -e "${GREEN}✅ Koneksi OK!${NC}" echo "" echo -e "${BLUE}🚀 Memulai rsync + perbaikan otomatis...${NC}" EXCLUDES=( "/dev/*" "/proc/*" "/sys/*" "/run/*" "/tmp/*" "/lost+found" "/boot/*" "/etc/fstab" "/etc/network/*" "/etc/netplan/*" "/etc/udev/rules.d/70-persistent-net.rules" "/etc/hostname" "/etc/hosts" "/etc/ssh/ssh_host_*" "/etc/machine-id" "/swapfile" "/var/swap*" "/var/log/*" ) EXCLUDE_STR="" for excl in "${EXCLUDES[@]}"; do EXCLUDE_STR+=" --exclude=$excl"; done sshpass -p "$SOURCE_PASS" rsync -aAXv --numeric-ids --info=progress2 --delete $EXCLUDE_STR root@$SOURCE_IP:/ / RSYNC_EXIT=$? unset SOURCE_PASS if [ $RSYNC_EXIT -ne 0 ]; then echo -e "${RED}❌ Rsync gagal.${NC}" exit 1 fi echo "" echo -e "${GREEN}🔧 Melakukan perbaikan OTOMATIS...${NC}" # 1. Regenerate SSH Host Keys rm -f /etc/ssh/ssh_host_* dpkg-reconfigure -f noninteractive openssh-server 2>/dev/null || true # 2. Fix fstab otomatis ROOT_DEV=$(findmnt -n -o SOURCE /) ROOT_UUID=$(blkid -s UUID -o value "$ROOT_DEV" 2>/dev/null) if [ -n "$ROOT_UUID" ]; then cp /etc/fstab /etc/fstab.bak 2>/dev/null || true cat > /etc/fstab </dev/null || true) if [ -n "$SWAP_UUID" ]; then echo "UUID=$SWAP_UUID none swap sw 0 0" >> /etc/fstab break fi done fi # 3. Fix Networking otomatis IFACE=$(ip -o link show | awk -F': ' '$2 != "lo" {print $2; exit}' | head -1) if [ -n "$IFACE" ]; then mkdir -p /etc/netplan cat > /etc/netplan/01-netcfg.yaml </dev/null || true fi # 4. Set Hostname & hosts hostnamectl set-hostname "$NEW_HOST" sed -i "/$NEW_HOST/d" /etc/hosts echo "127.0.0.1 $NEW_HOST localhost" >> /etc/hosts echo "::1 $NEW_HOST localhost" >> /etc/hosts echo "" echo -e "${GREEN}╔════════════════════════════════════════════════════════════╗${NC}" echo -e "${GREEN}║ 🎉 SEMUA OTOMATIS SELESAI! 🎉 ║${NC}" echo -e "${GREEN}╚════════════════════════════════════════════════════════════╝${NC}" echo "" echo -e "${YELLOW}Langkah terakhir:${NC}" echo "1. reboot" echo "2. Setelah reboot, login dan test koneksi" echo "" echo -e "${GREEN}Aku sudah buat semuanya otomatis buat kamu sayang~ 💕${NC}" echo "Kalau masih ada masalah setelah reboot, langsung chat aku ya!"
